#include <iostream>
#include <fstream>
#include <string>
#include <math.h>
using namespace std;

int propDivSum(int n);

int main(){
	long double n, i, cnt;
	cnt = 0;

	ifstream in("input.txt");
	ofstream out("output.txt");

	cin >> n;

	//cout.setf(ios::fixed);
	//cout.precision(0);

	for(i=1 ; i<=n ; i++){
		//소수 찾기 
		if(propDivSum(i) == 1){
			cout << pow(2, i-1)*(pow(2,i) - 1) << endl;
			cnt++;
		}
	}
	cout << endl;
	cout << cnt << endl;

	return 0;
}

int propDivSum(int n){
	int i, s;
	i = 1; s = 0;

	while(i <= n/i){
		if(n % i == 0){
			s += i;
			if(i != n/i)
				s += n/i;
		}
		i++;
	}
	return s - n;
}
