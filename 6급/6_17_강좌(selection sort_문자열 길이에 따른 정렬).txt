#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main(){
	ifstream in("input.txt");
	ofstream out("output.txt");
	
	string p[100], a, t;
	int n, i, j, s;
	int len, len1, len2;
	char m;
	m = ' '; s = 0; n = 0;
	a = "But we loved with a love that was more than love.";
	len = a.length();

	//split string with space (using concatenation)
	for(i=0 ; i<len ; i++){
		if(a[i] != ' ') p[n] += a[i];
		else n++;
	}

	//selection sort (fast version)
	//priority(0) : length of string
	//priority(1) : alphabetical order 
	for(i=0 ; i<n ; i++){
		s = i;
		for(j=i+1 ; j<n+1 ; j++){
			len1 = p[s].length();
			len2 = p[j].length();
			if(len1 > len2) s = j;
		}
		t = p[i]; p[i] = p[s]; p[s] = t;
	}

	//display result 
	for(i=0 ; i<=n ; i++)
		cout << p[i] << endl;

	return 0;
}
