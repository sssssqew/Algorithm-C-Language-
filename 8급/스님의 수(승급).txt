#include <iostream>
#include <fstream>
#include <cmath>
using namespace std;

int sum(int n);

int main(){
	int i, bowl, bab, gook;

	ifstream in("input.txt");
	ofstream out("output.txt");

	in >> bowl >> bab >> gook;

	for(i=1 ; i<=bowl-1 ; i++){
		if(bab * i == gook * (bowl - i)){
			out << bab * i << " " << i << " " << bowl - i << endl;
			break;
		}
	}

	in.close();
	out.close();

	return 0;
}

int sum(int n){
	int i, s;
	s = 0;
	for(i=1 ; i<n ; i++){
		if(n % i == 0 )
			s += i;
	}
	return s;
}

총그릇수 (밥그릇 + 국사발), 밥그릇 하나를 함께 쓰는 스님의 수, 
국사발 하나를 함께 쓰는 스님의 수가 입력으로 주어진다. 
스님의 수, 밥그릇 수, 국사발 수를 출력한다.
